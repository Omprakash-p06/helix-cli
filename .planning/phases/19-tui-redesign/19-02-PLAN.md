4. **Command filtering**:
```rust
pub fn filter_commands(commands: &[Command], filter: &str) -> Vec<Command> {
    if filter.is_empty() {
        return commands.to_vec();
    }
    let filter_lower = filter.to_lowercase();
    commands.iter()
        .filter(|c| c.name.to_lowercase().contains(&filter_lower) 
            || c.description.to_lowercase().contains(&filter_lower))
        .cloned()
        .collect()
}
```

```rust
fn handle_command_palette_input(key: KeyEvent, state: &mut TuiState) -> Option<TuiAction> {
    match key.code {
        KeyCode::Esc => Some(TuiAction::CloseCommandPalette),
        KeyCode::Char(c) => {
            state.command_palette.filter.push(c);
            None
        }
        KeyCode::Backspace => {
            state.command_palette.filter.pop();
            None
        }
        KeyCode::Up => {
            if state.command_palette.selected_index > 0 {
                state.command_palette.selected_index -= 1;
            }
            None
        }
        KeyCode::Down => {
            let max = state.command_palette.commands.len().saturating_sub(1);
            if state.command_palette.selected_index < max {
                state.command_palette.selected_index += 1;
            }
            None
        }
        KeyCode::Enter => {
            let cmd = state.command_palette.commands
                .get(state.command_palette.selected_index)?;
            execute_command(cmd)
        }
        _ => None,
    }
}
```

```rust
fn render_command_palette(f: &mut Frame, area: Rect, state: &TuiState) {
    if !state.command_palette.visible {
        return;
    }
    
    let filtered = filter_commands(
        &state.command_palette.commands, 
        &state.command_palette.filter
    );
    
    // Centered modal (50% width, 60% height)
    let modal_area = Rect::new(
        area.width / 4,
        area.height / 4,
        area.width * 3 / 4,
        area.height * 3 / 4,
    );
    
    let block = Block::default()
        .title("Command Palette")
        .borders(Borders::ALL)
        .border_style(Style::default().fg(state.current_theme().primary));
    
    // Filter input at top
    let filter_text = format!("Filter: {}", state.command_palette.filter);
    let filter = Paragraph::new(filter_text)
        .block(Block::default().borders(Borders::BOTTOM));
    
    // Command list
    let items: Vec<ListItem> = filtered.iter().enumerate().map(|(i, cmd)| {
        let marker = if i == state.command_palette.selected_index { ">" } else { " " };
        let text = format!("{} {} ({})", marker, cmd.name, cmd.shortcut.as_deref().unwrap_or(""));
        ListItem::new(text)
    }).collect();
    
    let list = List::new(items).block(Block::default());
    
    f.render_widget(block, modal_area);
    f.render_widget(filter, modal_area);
    f.render_widget(list, modal_area);
}
```
