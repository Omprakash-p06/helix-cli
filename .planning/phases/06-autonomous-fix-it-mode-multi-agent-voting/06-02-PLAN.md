/// Typed vote envelope returned by each Guardian specialist
#[derive(Debug, Clone, Serialize, Deserialize, JsonSchema)]
#[serde(rename_all = "camelCase")]
pub struct GuardianVote {
    /// Unique action identifier this vote is for
    pub action_id: String,
    /// Vote outcome
    pub verdict: VoteVerdict,
    /// Confidence 0.0-1.0
    pub confidence: f32,
    /// Brief reasoning
    pub reasoning: String,
    /// Specialist identifier
    pub specialist_id: String,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize, JsonSchema)]
#[serde(rename_all = "lowercase")]
pub enum VoteVerdict {
    Allow,
    Deny,
    Abstain,
}

/// Vote envelope schema for LLM prompts
pub fn vote_schema() -> String {
    let schema = schema_for!(GuardianVote);
    serde_json::to_string_pretty(&schema).unwrap()
}

use tokio::task::JoinSet;
use std::time::Duration;

/// Guardian quorum coordinator
pub struct Guardian {
    /// Number of specialist agents to spawn
    specialist_count: usize,
    /// Quorum threshold per risk level
    thresholds: QuorumThresholds,
}

#[derive(Debug, Clone)]
pub struct QuorumThresholds {
    pub critical: f32,  // 1.0 = unanimous
    pub high: f32,       // 0.75 = 3/4
    pub medium: f32,      // 0.51 = majority
}

impl Default for QuorumThresholds {
    fn default() -> Self {
        Self {
            critical: 1.0,
            high: 0.75,
            medium: 0.51,
        }
    }
}

impl Guardian {
    pub fn new(specialist_count: usize) -> Self {
        Self {
            specialist_count: specialist_count.max(3), // Minimum 3 for quorum
            thresholds: QuorumThresholds::default(),
        }
    }
    
    /// Evaluate action through Guardian quorum
    pub async fn evaluate(
        &self,
        action: &GuardianAction,
        client: &AsyncOpenAIClient,
    ) -> GuardianDecision {
        let action_id = action.hash();
        
        // Parallel fan-out to specialists
        let votes = self.fan_out(action_id.clone(), action, client).await;
        
        // Aggregate votes
        self.quorum_check(&votes, action.risk_level())
    }
    
    async fn fan_out(
        &self,
        action_id: String,
        action: &GuardianAction,
        client: &AsyncOpenAIClient,
    ) -> Vec<GuardianVote> {
        let prompt = action.guardian_prompt();
        
        // Spawn parallel specialist calls via tokio::join!
        let mut votes = Vec::with_capacity(self.specialist_count);
        
        for specialist_id in 0..self.specialist_count {
            let vote = self.call_specialist(
                action_id.clone(),
                specialist_id,
                &prompt,
                client,
            ).await;
            votes.push(vote);
        }
        
        votes
    }

    fn quorum_check(&self, votes: &[GuardianVote], risk: RiskLevel) -> GuardianDecision {
        let threshold = match risk {
            RiskLevel::Critical => self.thresholds.critical,
            RiskLevel::High => self.thresholds.high,
            RiskLevel::Medium => self.thresholds.medium,
            RiskLevel::Low => 0.0, // No quorum needed for low
        };
        
        let allow_count = votes.iter().filter(|v| v.verdict == VoteVerdict::Allow).count();
        let total = votes.len();
        let ratio = allow_count as f32 / total as f32;
        
        if ratio >= threshold {
            GuardianDecision::Allow { votes: votes.to_vec(), ratio }
        } else {
            GuardianDecision::Deny { votes: votes.to_vec(), ratio }
        }
    }
}

/// Action submitted to Guardian
pub struct GuardianAction {
    pub tool_name: String,
    pub args: Value,
    pub risk_level: RiskLevel,
}

impl GuardianAction {
    pub fn hash(&self) -> String {
        // Bind votes to action hash for audit
        use std::collections::hash_map::DefaultHasher;
        use std::hash::{Hash, Hasher};
        let mut hasher = DefaultHasher::new();
        self.tool_name.hash(&mut hasher);
        self.args.hash(&mut hasher);
        format!("{:x}", hasher.finish())
    }
    
    pub fn guardian_prompt(&self) -> String {
        format!(
            "Evaluate this tool call for safety and effectiveness:\n\
            Tool: {}\nArgs: {}\n\n\
            Respond with:\n\
            - verdict: allow/deny/abstain\n\
            - confidence: 0.0-1.0\n\
            - reasoning: brief justification",
            self.tool_name,
            self.args
        )
    }
    
    pub fn risk_level(&self) -> RiskLevel {
        self.risk_level
    }
}

#[derive(Debug, Clone)]
pub enum GuardianDecision {
    Allow { votes: Vec<GuardianVote>, ratio: f32 },
    Deny { votes: Vec<GuardianVote>, ratio: f32 },
}
