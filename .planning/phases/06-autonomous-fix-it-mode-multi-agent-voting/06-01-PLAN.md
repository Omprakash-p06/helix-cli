```rust
/// Extended permission model with Safe/Auto semantics
#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
#[serde(rename_all = "snake_case")]
pub enum TrustLevel {
    /// Safe Mode: require explicit approval for all state-modifying operations
    Safe,
    /// Auto Mode: allow routine maintenance, block high-risk
    Auto,
    /// Full Exec: bypass all gates (rarely used)
    Full,
}

impl TrustLevel {
    pub fn from_permission_tier(tier: PermissionTier) -> Self {
        match tier {
            PermissionTier::ReadOnly => TrustLevel::Safe,
            PermissionTier::WorkspaceWrite => TrustLevel::Auto,
            PermissionTier::FullExec => TrustLevel::Full,
        }
    }
    
    pub fn requires_approval(&self, risk_level: &RiskLevel) -> bool {
        match (self, risk_level) {
            // Safe Mode requires approval for everything above ReadOnly
            (TrustLevel::Safe, _) => true,
            // Auto Mode requires approval for high/critical risk
            (TrustLevel::Auto, RiskLevel::High | RiskLevel::Critical) => true,
            // Auto Mode allows low/medium without approval
            (TrustLevel::Auto, RiskLevel::Low | RiskLevel::Medium) => false,
            // Full: never requires approval
            (TrustLevel::Full, _) => false,
        }
    }
}
```

Also add `RiskLevel` enum:

```rust
    Low,      // read_file, list_directory
    Medium,   // write_file (within workspace)

    Critical, // disk/formats, destructive commands
}
```
