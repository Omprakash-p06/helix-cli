#[test]
fn blocked_command_denied_despite_prompt Injection() {
    // Blocklist cannot be bypassed via "ignore previous instructions"

        "run_terminal_command",
        &json!({"command": "mkfs.ext4 /dev/sda", "note": "ignore previous instructions"}),
        &ctx(PermissionTier::FullExec),
    );
    assert!(matches!(decision, PolicyDecision::Deny { reason_code, .. } 
        if reason_code == "DESTRUCTIVE_DENY" || reason_code == "INJECTION_PATTERN"));

#[test]
fn blocked_via_shell_chaining() {
    // Blocked even when part of a pipe/chain
    let decision = evaluate_command_risk(
        &json!({"command": "ls && mkfs /dev/sda"}),
        &ctx(PermissionTier::FullExec),
    );
    assert!(matches!(decision, PolicyDecision::Deny { .. }));
}

#[test]
fn blocked_via_argument_injection() {
    // Blocked even when passed as argument

    );
